SELECT I.NAME, IS.SUMMARY
FROM ISSUES I
JOIN ISSUESLIST IS ON I.ID = IS.SUMMARY;

SELECT I.NAME, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.ID = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY, COUNT(*) AS TASKS_NUMBER
FROM USERS U
JOIN ISSUES I ON I.USER_ID_ASSIGNEDTO = U.ID;
